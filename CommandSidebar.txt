  <script>
    // ===== 状態 =====
    var isRunning = false;       // ループ実行中
    var cancelRequested = false; // キャンセル要求フラグ（runNext ループを止める）
    var runId = 0; // 実行世代。onStart/onCancel で進め、古いループの遅延レスポンスを無効化する
    var pendingResume = false;   // 再開待ち（未完了ジョブ検出済み・再開指示待ち）

    // ===== 進捗表示 =====
    function updateProgress(data) {
      if (!data) return;

      var total = (data.totalRows !== undefined && data.totalRows !== null) ? data.totalRows : 0;
      var processed = (data.processedRows !== undefined && data.processedRows !== null) ? data.processedRows : 0;

      document.getElementById('progressText').textContent = processed + ' / ' + total;
      var pct = (total > 0) ? Math.min(100, Math.round((processed / total) * 100)) : 0;
      document.getElementById('progressBar').style.width = pct + '%';

      if (data.successRows !== undefined && data.successRows !== null) {
        document.getElementById('successCount').textContent = data.successRows;
      }
      if (data.failedRows !== undefined && data.failedRows !== null) {
        document.getElementById('failedCount').textContent = data.failedRows;
      }
    }

    // ===== ログ =====
    var lastMessage = '';
    function addLogEntry(message, type) {
      if (!message) return;
      if (message === lastMessage) return;
      lastMessage = message;

      var logContainer = document.getElementById('logEntries');
      var entry = document.createElement('div');
      entry.className = 'log-entry ' + (type || 'info');

      var timestamp = new Date().toLocaleTimeString('ja-JP');
      var span = document.createElement('span');
      span.className = 'timestamp';
      span.textContent = timestamp;
      entry.appendChild(span);
      entry.appendChild(document.createTextNode(message));

      logContainer.insertBefore(entry, logContainer.firstChild);
      while (logContainer.children.length > 50) {
        logContainer.removeChild(logContainer.lastChild);
      }
    }

    // ===== 完了バナー =====
    function showCompletionBanner(result, failed) {
      var banner = document.getElementById('completionBanner');
      var title = document.getElementById('completionTitle');
      var msg = document.getElementById('completionMessage');
      if (failed) {
        banner.classList.add('failed');
        document.getElementById('headerTitle').textContent = '⛔ 処理停止';
        title.textContent = '⛔ 処理が停止しました';
        msg.textContent = (result && result.message) ? result.message : 'エラーにより処理を停止しました。';
      } else {
        document.getElementById('headerTitle').textContent = '✅ サイドパネルでバッチ翻訳';
        title.textContent = '✅ 処理完了！';
        var s = (result && result.successRows !== undefined) ? result.successRows : 0;
        var f = (result && result.failedRows !== undefined) ? result.failedRows : 0;
        msg.textContent = '成功 ' + s + ' 件 / 失敗 ' + f + ' 件。サイドバーを閉じてください。';
      }
      banner.classList.add('show');
    }

    // ===== UI 状態切替 =====
    function setRunningUi(running) {
      isRunning = running;
      document.getElementById('startBtn').disabled = running;
      document.getElementById('cancelBtn').disabled = !running;
      document.getElementById('startRow').disabled = running;
      document.getElementById('endRow').disabled = running;
      document.getElementById('chunkSize').disabled = running;
    }

    function stopLoop() {
      setRunningUi(false);
    }

    // ===== 司令塔ループ（runNext 再帰コールバック）=====
    function runNext(result, myRun) {
      // 古い実行世代の遅延レスポンスは UI に触れず破棄（キャンセル→再開の二重ループ防止）
      if (myRun !== runId) return;

      // 最優先でキャンセル判定（現役世代のみ）
      if (cancelRequested) { addLogEntry('キャンセルしました', 'warning'); stopLoop(); return; }

      if (result) {
        if (result.ok === false && result.message) addLogEntry(result.message, 'error');
        updateProgress(result);
        if (result.skipCount && result.skipCount > 0) addLogEntry('空行を ' + result.skipCount + ' 行スキップしました', 'warning');
      }

      if (result && result.fatal) { addLogEntry('停止: ' + (result.message || '致命的なエラー'), 'error'); showCompletionBanner(result, true); stopLoop(); return; }

      if (result && result.done) { showCompletionBanner(result, false); addLogEntry('✅ 全チャンクの処理が完了しました', 'success'); stopLoop(); return; }

      if (result && result.busy) {
        addLogEntry('実行中のため待機します…', 'info');
        setTimeout(function() { if (myRun !== runId) return; if (cancelRequested) { stopLoop(); return; } callProcessChunk(myRun); }, 2000);
        return;
      }

      if (result && result.ok === false && result.retryable) {
        addLogEntry('チャンク失敗（' + (result.message || '') + '）。3秒後に再試行します', 'warning');
        setTimeout(function() { if (myRun !== runId) return; if (cancelRequested) { stopLoop(); return; } callProcessChunk(myRun); }, 3000);
        return;
      }

      if (result && result.ok === false) { addLogEntry('停止: 想定外の応答 ' + (result.message || ''), 'error'); showCompletionBanner(result, true); stopLoop(); return; }

      callProcessChunk(myRun);
    }

    function callProcessChunk(myRun) {
      google.script.run
        .withSuccessHandler(function(result) { runNext(result, myRun); })
        .withFailureHandler(function(error) { onChunkError(error, myRun); })
        .ebApiSbProcessChunk();
    }

    // クライアント側の通信エラー（withFailureHandler）
    function onChunkError(error, myRun) {
      if (myRun !== runId) return; // 古い世代は無視
      var msg = (error && error.message) ? error.message : String(error);
      addLogEntry('通信エラー: ' + msg, 'error');
      stopLoop();
    }

    // ===== 選択範囲を取り込む =====
    function applySelection() {
      google.script.run
        .withSuccessHandler(function(sel) {
          if (sel && sel.ok) {
            document.getElementById('startRow').value = sel.startRow;
            document.getElementById('endRow').value = sel.endRow;
            addLogEntry('選択範囲を取り込みました（' + sel.startRow + '〜' + sel.endRow + '行）', 'success');
          } else {
            addLogEntry((sel && sel.message) ? sel.message : '選択範囲を取得できませんでした', 'warning');
          }
        })
        .withFailureHandler(function(error) {
          addLogEntry('選択範囲の取得エラー: ' + ((error && error.message) ? error.message : String(error)), 'error');
        })
        .ebApiSbGetSelection();
    }

    // ===== 開始ボタン =====
    function onStart() {
      if (isRunning) return;

      if (pendingResume) {
        // 再開モード: ebApiSbStart を呼ばず既存ジョブで runNext ループを直接再開
        cancelRequested = false;
        var myResume = ++runId;
        pendingResume = false;
        document.getElementById('startBtn').textContent = '開始';
        setRunningUi(true);
        document.getElementById('completionBanner').classList.remove('show', 'failed');
        addLogEntry('翻訳を再開します', 'info');
        runNext(undefined, myResume);
        return;
      }

      var startRow = parseInt(document.getElementById('startRow').value, 10);
      var endRow = parseInt(document.getElementById('endRow').value, 10);
      var chunkSize = parseInt(document.getElementById('chunkSize').value, 10);

      if (!startRow || startRow < 1) { addLogEntry('開始行が不正です', 'error'); return; }
      if (!endRow || endRow < startRow) { addLogEntry('終了行は開始行以上にしてください', 'error'); return; }
      if (!chunkSize || chunkSize < 1) { chunkSize = 15; }
      if (chunkSize > 15) { chunkSize = 15; addLogEntry('チャンクサイズを上限15に調整しました', 'warning'); }

      cancelRequested = false;
      var myRun = ++runId; // 入力値チェック通過後に世代を発行
      setRunningUi(true);
      document.getElementById('completionBanner').classList.remove('show', 'failed');
      addLogEntry('翻訳を開始します（' + startRow + '行〜' + endRow + '行, チャンク' + chunkSize + '行）', 'info');

      var config = { startRow: startRow, endRow: endRow, chunkSize: chunkSize };
      google.script.run
        .withSuccessHandler(function(result) { onStartSuccess(result, myRun); })
        .withFailureHandler(function(error) { onStartError(error, myRun); })
        .ebApiSbStart(config);
    }

    function onStartSuccess(result, myRun) {
      if (myRun !== runId) return;
      if (result && result.ok === false) { addLogEntry('開始できません: ' + (result.message || ''), 'error'); stopLoop(); initState(); return; }
      addLogEntry((result && result.message) ? result.message : 'ジョブを開始しました', 'info');
      updateProgress(result);
      runNext(undefined, myRun); // 初回キック
    }

    function onStartError(error, myRun) {
      if (myRun !== runId) return;
      var msg = (error && error.message) ? error.message : String(error);
      addLogEntry('開始エラー: ' + msg, 'error');
      stopLoop();
    }

    // ===== キャンセルボタン =====
    function onCancel() {
      cancelRequested = true;
      runId++; // 進行中ループを無効化（古い世代の遅延レスポンスを破棄）
      pendingResume = false;
      document.getElementById('startBtn').textContent = '開始';
      addLogEntry('キャンセルを要求しました…', 'warning');
      document.getElementById('cancelBtn').disabled = true;
      google.script.run
        .withSuccessHandler(function(result){ addLogEntry((result && result.message) ? result.message : '中止しました', 'warning'); stopLoop(); })
        .withFailureHandler(function(error){ var msg = (error && error.message) ? error.message : String(error); addLogEntry('キャンセル失敗。再試行してください: ' + msg, 'error'); pendingResume = true; isRunning = false; document.getElementById('startBtn').textContent = '再開'; document.getElementById('startBtn').disabled = false; document.getElementById('cancelBtn').disabled = false; })
        .ebApiSbCancel();
    }

    function closeSidebar() {
      google.script.host.close();
    }

    // ===== 初期化：既存ジョブがあれば状態を表示（再開導線）=====
    function initState() {
      applySelection();
      google.script.run
        .withSuccessHandler(function(state) {
          if (isRunning) return; // 実行中は状態同期で UI/フラグを壊さない
          if (state && state.savedChunkSize) {
            var csEl = document.getElementById('chunkSize');
            if (csEl) csEl.value = state.savedChunkSize;
          }
          if (state && state.ok) {
            if (state.totalRows) updateProgress(state);
            if (state.status === 'running') {
              pendingResume = true;
              document.getElementById('startBtn').textContent = '再開';
              document.getElementById('startBtn').disabled = false;
              document.getElementById('cancelBtn').disabled = false;
              document.getElementById('startRow').disabled = true;
              document.getElementById('endRow').disabled = true;
              document.getElementById('chunkSize').disabled = true;
              addLogEntry('未完了のジョブがあります。「再開」で続きから、「キャンセル」でリセットできます', 'warning');
            }
          }
        })
        .withFailureHandler(function() { /* ジョブなしは無視 */ })
        .ebApiSbGetState();
    }

    addLogEntry('サイドパネルを初期化しました', 'info');
    initState();
  </script>
