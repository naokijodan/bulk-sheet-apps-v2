  if (category === 'game') {

    lines.push('- ストレージ容量: GB単位で記入（例: 32GB, 825GB）。不明ならNA。');

    lines.push('- サイズ/番手: 型番に含まれる数字（例: 2500, C3000, 103）をそのまま記入。');
